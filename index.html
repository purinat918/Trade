input double InpOverbought = 65.0; input double InpOversold = 35.0; input int InpStartHour = 19; input int InpEndHour = 23; input int InpTimeOffset = 4; // ⏱️ เพิ่มเมนูตั้งค่าบวกเวลาตรงนี้ double CallBuffer[]; double PutBuffer[]; int rsi_handle; datetime last_alert_time = 0; int OnInit() { SetIndexBuffer(0, CallBuffer, INDICATOR_DATA); PlotIndexSetInteger(0, PLOT_ARROW, 233); SetIndexBuffer(1, PutBuffer, INDICATOR_DATA); PlotIndexSetInteger(1, PLOT_ARROW, 234); rsi_handle = iRSI(_Symbol, _Period, InpRSI_Period, PRICE_CLOSE); if(rsi_handle == INVALID_HANDLE) return(INIT_FAILED); // แสดงข้อความว่าชดเชยเวลาไปกี่ชั่วโมงบนหน้ากราฟด้วย Comment("\n\n 🔥 โหมดเร่งโตดุดัน (Aggressive) ทำงานแล้ว!\n [สแกนช่วงเวลา 19:00 - 23:00 น. | Offset: +", InpTimeOffset, " ชม.]\n สแกนคู่เงิน: ", _Symbol); return(INIT_SUCCEEDED); } void OnDeinit(const int reason) { Comment(""); } int OnCalculate(const int rates_total, const int prev_calculated, const datetime &time[], const double &open[], const double &high[], const double &low[], const double &close[], const long &tick_volume[], const long &volume[], const int &spread[]) { if(rates_total < 3) return(0); double rsi_buffer[]; ArraySetAsSeries(rsi_buffer, true); if(CopyBuffer(rsi_handle, 0, 0, rates_total, rsi_buffer) <= 0) return 0; ArraySetAsSeries(open, true); ArraySetAsSeries(close, true); ArraySetAsSeries(high, true); ArraySetAsSeries(low, true); ArraySetAsSeries(CallBuffer, true); ArraySetAsSeries(PutBuffer, true); ArraySetAsSeries(time, true); int limit = rates_total - prev_calculated; if(limit == 0) limit = 1; if(prev_calculated == 0) { limit = rates_total - 2; ArrayInitialize(CallBuffer, EMPTY_VALUE); ArrayInitialize(PutBuffer, EMPTY_VALUE); } MqlDateTime dt; TimeToStruct(TimeLocal(), dt); for(int i = limit; i >= 1; i--) { MqlDateTime bar_dt; // 💡 เปลี่ยนมาใช้ตัวแปร InpTimeOffset ทำให้ปรับเปลี่ยนหน้ากราฟได้เลย TimeToStruct(time[i] + InpTimeOffset*3600, bar_dt); double lower_shadow = MathMin(open[i], close[i]) - low[i]; bool bull_pin = (lower_shadow >= body * 1.5) && (upper_shadow <= body); bool bear_pin = (upper_shadow >= body * 1.5) && (lower_shadow <= body); bool bull_eng = (close[i+1] < open[i+1]) && (close[i] > open[i]) && (close[i] >= open[i+1]); bool bear_eng = (close[i+1] > open[i+1]) && (close[i] < open[i]) && (close[i] <= open[i+1]); bool call_signal = (bull_pin || bull_eng) && (rsi_buffer[i] < InpOversold || rsi_buffer[i+1] < InpOversold); bool put_signal = (bear_pin || bear_eng) && (rsi_buffer[i] > InpOverbought || rsi_buffer[i+1] > InpOverbought); if(call_signal) CallBuffer[i] = low[i]; else CallBuffer[i] = EMPTY_VALUE; if(put_signal) PutBuffer[i] = high[i]; else PutBuffer[i] = EMPTY_VALUE; } if(dt.hour >= InpStartHour && dt.hour < InpEndHour) { if(time[1] != last_alert_time) { if(CallBuffer[1] != EMPTY_VALUE) { SendNotification("🔥 [ลุยดุดัน] CALL: " + _Symbol + " มาแล้ว!"); last_alert_time = time[1]; } else if(PutBuffer[1] != EMPTY_VALUE) { SendNotification("🔥 [ลุยดุดัน] PUT: " + _Symbol + " มาแล้ว!"); last_alert_time = time[1]; } } } return(rates_total); }
